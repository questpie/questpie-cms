# Getting Started with QUESTPIE CMS

QUESTPIE CMS is a headless, extensible CMS framework for TypeScript. This guide will get you up and running in minutes.

## What is QUESTPIE CMS?

QUESTPIE provides:

- **Backend CMS** (`questpie`) - Type-safe data layer with Drizzle ORM
- **Admin UI** (`@questpie/admin`) - Beautiful, config-driven admin interface
- **Framework Adapters** - Hono, Elysia, Next.js

## Quick Start

### 1. Create a New Project

```bash
# Create a new TanStack Start project
npx create-t3-app@latest my-cms --tailwind

# Or use the example project
git clone https://github.com/questpie/questpie-cms
cd questpie-cms/examples/tanstack-barbershop
bun install
```

### 2. Install Dependencies

```bash
bun add questpie @questpie/admin @questpie/hono
bun add @questpie/tanstack-query @tanstack/react-query
bun add drizzle-orm postgres
bun add -D drizzle-kit
```

### 3. Define Your CMS

```typescript
// src/questpie/server/cms.ts
import { q, text, integer, boolean, datetime } from "questpie";
import { adminModule } from "@questpie/admin/server";

export const cms = q({ name: "my-cms" })
  .use(adminModule)  // Adds auth and assets
  .collections({
    posts: q.collection("posts")
      .fields({
        title: text("title").notNull(),
        content: text("content"),
        published: boolean("published").default(false),
        publishedAt: datetime("publishedAt"),
      })
      .validation({
        exclude: { id: true, createdAt: true, updatedAt: true },
        refine: {
          title: (s) => s.min(1).max(200),
        },
      }),
  })
  .auth({
    emailAndPassword: { enabled: true },
    secret: process.env.AUTH_SECRET,
  })
  .build({
    db: { url: process.env.DATABASE_URL },
  });

export type AppCMS = typeof cms;
```

### 4. Configure Admin UI

```typescript
// src/questpie/admin/admin.ts
import { qa, adminModule } from "@questpie/admin/client";
import type { AppCMS } from "../server/cms";

export const admin = qa<AppCMS>()
  .use(adminModule)
  .branding({ name: "My CMS" })
  .collections({
    posts: qa.collection("posts")
      .meta({ label: "Posts" })
      .fields(({ r }) => ({
        title: r.text({ label: "Title", required: true }),
        content: r.richText({ label: "Content" }),
        published: r.switch({ label: "Published" }),
        publishedAt: r.datetime({ label: "Published At" }),
      }))
      .list(({ v, f }) => v.table({
        columns: [f.title, f.published, f.publishedAt],
      }))
      .form(({ v, f }) => v.form({
        fields: [f.title, f.content, f.published, f.publishedAt],
      })),
  })
  .sidebar(
    qa.sidebar<AppCMS>()
      .section("content", (s) =>
        s.title("Content").items([
          { type: "collection", collection: "posts" },
        ])
      )
  );
```

### 5. Mount the API

```typescript
// src/server.ts (Hono)
import { Hono } from "hono";
import { createHonoAdapter } from "@questpie/hono";
import { cms } from "./questpie/server/cms";

const app = new Hono();
app.route("/api/cms", createHonoAdapter(cms));

export default app;
```

### 6. Mount the Admin

```tsx
// src/routes/admin.tsx
import { Admin, AdminLayoutProvider } from "@questpie/admin/client";
import { admin } from "@/questpie/admin/admin";
import { cmsClient } from "@/lib/cms-client";

const adminInstance = Admin.from(admin);

export default function AdminPage() {
  return (
    <AdminLayoutProvider
      admin={adminInstance}
      client={cmsClient}
      basePath="/admin"
    >
      {/* Admin UI renders here */}
    </AdminLayoutProvider>
  );
}
```

### 7. Run Migrations

```bash
# Generate migration
bun run drizzle-kit generate

# Apply migration
bun run drizzle-kit migrate
```

### 8. Start the Dev Server

```bash
bun run dev
```

Visit `http://localhost:3000/admin` to access the admin UI.

## Project Structure

Recommended structure for QUESTPIE projects:

```
src/
├── questpie/
│   ├── server/           # Backend CMS
│   │   ├── cms.ts        # Main CMS instance
│   │   ├── collections/  # Collection definitions
│   │   └── globals/      # Global definitions
│   │
│   └── admin/            # Admin UI
│       ├── admin.ts      # Main admin config
│       ├── builder.ts    # Shared builder
│       ├── collections/  # Collection UI configs
│       └── sidebar.ts    # Navigation
│
├── routes/
│   ├── api/cms.ts        # CMS API routes
│   └── admin.tsx         # Admin UI route
│
└── lib/
    └── cms-client.ts     # CMS client instance
```
