# Global UI Configuration

Globals are singleton documents - settings, configurations, or other data that exists as a single instance. The admin UI for globals is similar to collections but without list views.

## Creating a Global Config

```typescript
import { builder } from "./builder";
import { GearIcon } from "@phosphor-icons/react";

export const siteSettingsAdmin = builder
  .global("siteSettings")      // Must match backend global name
  .meta({                      // UI metadata
    label: "Site Settings",
    icon: GearIcon,
  })
  .fields(({ r }) => ({...}))  // Field configurations
  .form(({ v, f }) => {...});  // Edit form
```

## `.meta()` - UI Metadata

```typescript
.meta({
  label: { en: "Site Settings", sk: "Nastavenia stranky" },
  icon: GearIcon,
  description: "Configure global site settings",
})
```

## `.fields()` - Field Configurations

Define global fields the same way as collections:

```typescript
.fields(({ r }) => ({
  siteName: r.text({
    label: { en: "Site Name", sk: "Nazov stranky" },
    maxLength: 100,
    required: true,
  }),

  siteDescription: r.textarea({
    label: { en: "Description", sk: "Popis" },
    rows: 3,
  }),

  logo: r.upload({
    label: { en: "Logo", sk: "Logo" },
    accept: ["image/*"],
  }),

  contactEmail: r.email({
    label: { en: "Contact Email", sk: "Kontaktny email" },
  }),

  socialLinks: r.array({
    label: { en: "Social Links", sk: "Socialne siete" },
    item: ({ r }) => ({
      platform: r.select({
        label: "Platform",
        options: [
          { label: "Facebook", value: "facebook" },
          { label: "Instagram", value: "instagram" },
          { label: "Twitter", value: "twitter" },
        ],
      }),
      url: r.text({ label: "URL" }),
    }),
  }),

  maintenanceMode: r.switch({
    label: { en: "Maintenance Mode", sk: "Udrzba" },
    description: {
      en: "When enabled, shows maintenance page to visitors",
      sk: "Ked je zapnute, zobrazuje udrzbu navstevnikom",
    },
  }),
}))
```

## `.form()` - Edit Form

Configure the global edit form:

```typescript
.form(({ v, f }) => v.form({
  sections: [
    {
      title: { en: "General", sk: "Vseobecne" },
      fields: [f.siteName, f.siteDescription, f.logo],
    },
    {
      title: { en: "Contact", sk: "Kontakt" },
      fields: [f.contactEmail, f.socialLinks],
    },
    {
      title: { en: "Advanced", sk: "Pokrocile" },
      collapsible: true,
      fields: [f.maintenanceMode],
    },
  ],
}))
```

## Complete Example

```typescript
// admin/globals/site-settings.ts
import { GearIcon } from "@phosphor-icons/react";
import { builder } from "../builder";

export const siteSettingsAdmin = builder
  .global("siteSettings")
  .meta({
    label: { en: "Site Settings", sk: "Nastavenia" },
    icon: GearIcon,
  })
  .fields(({ r }) => ({
    siteName: r.text({
      label: { en: "Site Name", sk: "Nazov" },
      required: true,
    }),
    tagline: r.text({
      label: { en: "Tagline", sk: "Slogan" },
    }),
    logo: r.upload({
      label: { en: "Logo", sk: "Logo" },
      accept: ["image/png", "image/svg+xml"],
    }),
    favicon: r.upload({
      label: { en: "Favicon", sk: "Favicon" },
      accept: ["image/x-icon", "image/png"],
    }),
    contactEmail: r.email({
      label: { en: "Contact Email", sk: "Kontaktny email" },
    }),
    contactPhone: r.text({
      label: { en: "Phone", sk: "Telefon" },
    }),
    address: r.textarea({
      label: { en: "Address", sk: "Adresa" },
      rows: 3,
    }),
    businessHours: r.object({
      label: { en: "Business Hours", sk: "Otvaracie hodiny" },
      fields: ({ r }) => ({
        weekdays: r.text({ label: { en: "Weekdays", sk: "Pracovne dni" } }),
        saturday: r.text({ label: { en: "Saturday", sk: "Sobota" } }),
        sunday: r.text({ label: { en: "Sunday", sk: "Nedela" } }),
      }),
    }),
    socialLinks: r.array({
      label: { en: "Social Media", sk: "Socialne siete" },
      item: ({ r }) => ({
        platform: r.select({
          label: "Platform",
          options: [
            { label: "Facebook", value: "facebook" },
            { label: "Instagram", value: "instagram" },
            { label: "Twitter", value: "twitter" },
            { label: "LinkedIn", value: "linkedin" },
          ],
        }),
        url: r.text({ label: "URL" }),
      }),
      maxItems: 5,
    }),
  }))
  .form(({ v, f }) => v.form({
    tabs: [
      {
        id: "general",
        label: { en: "General", sk: "Vseobecne" },
        fields: [f.siteName, f.tagline, f.logo, f.favicon],
      },
      {
        id: "contact",
        label: { en: "Contact", sk: "Kontakt" },
        fields: [
          f.contactEmail,
          f.contactPhone,
          f.address,
          f.businessHours,
        ],
      },
      {
        id: "social",
        label: { en: "Social", sk: "Socialne siete" },
        fields: [f.socialLinks],
      },
    ],
  }));
```

## Adding Globals to Admin

Register globals in your main admin config:

```typescript
// admin/admin.ts
import { qa, adminModule } from "@questpie/admin/client";
import { siteSettingsAdmin } from "./globals/site-settings";

export const admin = qa()
  .use(adminModule)
  .globals({
    siteSettings: siteSettingsAdmin,
  });
```

## Adding to Sidebar

Add a link to the global in your sidebar:

```typescript
// admin/sidebar.ts
.section("settings", (s) =>
  s
    .title("Settings")
    .icon(GearIcon)
    .items([
      { type: "global", global: "siteSettings", icon: GearIcon },
    ])
)
```

## Accessing Global Data

Use the `useGlobal` hook to read global data:

```typescript
import { useGlobal, useGlobalUpdate } from "@questpie/admin/client";

function SettingsPage() {
  const { data: settings, isLoading } = useGlobal("siteSettings");
  const updateSettings = useGlobalUpdate("siteSettings");

  if (isLoading) return <Loading />;

  return (
    <div>
      <h1>{settings.siteName}</h1>
      <button onClick={() => updateSettings.mutate({ siteName: "New Name" })}>
        Update
      </button>
    </div>
  );
}
```
