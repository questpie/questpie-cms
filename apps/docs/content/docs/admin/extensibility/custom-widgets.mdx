# Custom Widgets

Create custom dashboard widgets to display analytics, summaries, and quick actions.

## Widget Structure

```typescript
import { widget } from "@questpie/admin/client";

const myWidget = widget("myWidget", {
  // Required: React component
  component: MyWidgetComponent,

  // Required: TypeScript config type
  config: {} as MyWidgetConfig,

  // Optional: Default configuration
  defaultConfig: { ... },
});
```

## Example: Analytics Card

A simple analytics card showing a metric with comparison.

### 1. Define Config Type

```typescript
interface AnalyticsCardConfig {
  label: string;
  value: () => Promise<number>;
  previousValue?: () => Promise<number>;
  format?: "number" | "currency" | "percent";
  icon?: React.ComponentType;
}
```

### 2. Create Widget Component

```tsx
function AnalyticsCardWidget({ config }: { config: AnalyticsCardConfig }) {
  const [value, setValue] = useState<number | null>(null);
  const [prevValue, setPrevValue] = useState<number | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function load() {
      setLoading(true);
      const [current, previous] = await Promise.all([
        config.value(),
        config.previousValue?.(),
      ]);
      setValue(current);
      setPrevValue(previous ?? null);
      setLoading(false);
    }
    load();
  }, [config]);

  const formatValue = (v: number) => {
    switch (config.format) {
      case "currency":
        return `$${v.toLocaleString()}`;
      case "percent":
        return `${v}%`;
      default:
        return v.toLocaleString();
    }
  };

  const percentChange = prevValue
    ? ((value! - prevValue) / prevValue) * 100
    : null;

  const Icon = config.icon;

  return (
    <Card>
      <CardContent className="p-6">
        <div className="flex items-center justify-between">
          <div>
            <p className="text-sm text-muted-foreground">{config.label}</p>
            {loading ? (
              <Skeleton className="h-8 w-20" />
            ) : (
              <p className="text-2xl font-bold">{formatValue(value!)}</p>
            )}
          </div>
          {Icon && (
            <div className="rounded-full bg-primary/10 p-3">
              <Icon className="h-5 w-5 text-primary" />
            </div>
          )}
        </div>

        {percentChange !== null && (
          <div className="mt-2 flex items-center gap-1">
            {percentChange >= 0 ? (
              <TrendingUp className="h-4 w-4 text-green-500" />
            ) : (
              <TrendingDown className="h-4 w-4 text-red-500" />
            )}
            <span
              className={cn(
                "text-sm",
                percentChange >= 0 ? "text-green-500" : "text-red-500"
              )}
            >
              {Math.abs(percentChange).toFixed(1)}%
            </span>
            <span className="text-sm text-muted-foreground">vs last period</span>
          </div>
        )}
      </CardContent>
    </Card>
  );
}
```

### 3. Create Widget Definition

```typescript
export const analyticsCardWidget = widget("analyticsCard", {
  component: AnalyticsCardWidget,
  config: {} as AnalyticsCardConfig,
  defaultConfig: {
    format: "number",
  },
});
```

### 4. Register and Use

```typescript
const admin = qa()
  .use(adminModule)
  .widgets({ analyticsCard: analyticsCardWidget })
  .dashboard(({ w }) => ({
    widgets: [
      w.analyticsCard({
        label: "Total Revenue",
        value: async () => {
          const result = await db.query("SELECT SUM(amount) FROM orders");
          return result.sum;
        },
        previousValue: async () => {
          const result = await db.query("SELECT SUM(amount) FROM orders WHERE date < NOW() - INTERVAL '30 days'");
          return result.sum;
        },
        format: "currency",
        icon: DollarSign,
      }),
    ],
  }));
```

## Example: Activity Feed

A real-time activity feed widget.

```typescript
interface ActivityFeedConfig {
  title: string;
  fetchActivities: () => Promise<Activity[]>;
  maxItems?: number;
  showTimestamp?: boolean;
}

interface Activity {
  id: string;
  type: "create" | "update" | "delete";
  collection: string;
  itemTitle: string;
  user: { name: string; avatar?: string };
  timestamp: Date;
}

function ActivityFeedWidget({ config }: { config: ActivityFeedConfig }) {
  const [activities, setActivities] = useState<Activity[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    config.fetchActivities().then((data) => {
      setActivities(data.slice(0, config.maxItems ?? 10));
      setLoading(false);
    });
  }, [config]);

  const getIcon = (type: Activity["type"]) => {
    switch (type) {
      case "create": return <Plus className="h-4 w-4 text-green-500" />;
      case "update": return <Pencil className="h-4 w-4 text-blue-500" />;
      case "delete": return <Trash className="h-4 w-4 text-red-500" />;
    }
  };

  return (
    <Card>
      <CardHeader>
        <CardTitle>{config.title}</CardTitle>
      </CardHeader>
      <CardContent>
        {loading ? (
          <div className="space-y-4">
            {[...Array(5)].map((_, i) => (
              <Skeleton key={i} className="h-12" />
            ))}
          </div>
        ) : (
          <div className="space-y-4">
            {activities.map((activity) => (
              <div key={activity.id} className="flex items-start gap-3">
                <Avatar className="h-8 w-8">
                  <AvatarImage src={activity.user.avatar} />
                  <AvatarFallback>
                    {activity.user.name.charAt(0)}
                  </AvatarFallback>
                </Avatar>
                <div className="flex-1">
                  <p className="text-sm">
                    <span className="font-medium">{activity.user.name}</span>
                    {" "}
                    {activity.type === "create" && "created"}
                    {activity.type === "update" && "updated"}
                    {activity.type === "delete" && "deleted"}
                    {" "}
                    <span className="font-medium">{activity.itemTitle}</span>
                    {" in "}
                    <span className="text-muted-foreground">
                      {activity.collection}
                    </span>
                  </p>
                  {config.showTimestamp && (
                    <p className="text-xs text-muted-foreground">
                      {formatDistanceToNow(activity.timestamp, { addSuffix: true })}
                    </p>
                  )}
                </div>
                {getIcon(activity.type)}
              </div>
            ))}
          </div>
        )}
      </CardContent>
    </Card>
  );
}

export const activityFeedWidget = widget("activityFeed", {
  component: ActivityFeedWidget,
  config: {} as ActivityFeedConfig,
  defaultConfig: {
    maxItems: 10,
    showTimestamp: true,
  },
});
```

## Example: Quick Links

A widget with quick action buttons.

```typescript
interface QuickLinksConfig {
  title: string;
  links: {
    label: string;
    href: string;
    icon?: React.ComponentType;
    variant?: "default" | "outline" | "ghost";
  }[];
}

function QuickLinksWidget({ config }: { config: QuickLinksConfig }) {
  return (
    <Card>
      <CardHeader>
        <CardTitle>{config.title}</CardTitle>
      </CardHeader>
      <CardContent>
        <div className="grid gap-2">
          {config.links.map((link, i) => {
            const Icon = link.icon;
            return (
              <Button
                key={i}
                variant={link.variant ?? "outline"}
                asChild
                className="justify-start"
              >
                <a href={link.href}>
                  {Icon && <Icon className="mr-2 h-4 w-4" />}
                  {link.label}
                </a>
              </Button>
            );
          })}
        </div>
      </CardContent>
    </Card>
  );
}

export const quickLinksWidget = widget("quickLinks", {
  component: QuickLinksWidget,
  config: {} as QuickLinksConfig,
});
```

## Dashboard Layout

Widgets can be arranged in a grid:

```typescript
.dashboard(({ w }) => ({
  widgets: [
    w.analyticsCard({ ... }),
    w.analyticsCard({ ... }),
    w.activityFeed({ ... }),
    w.quickLinks({ ... }),
  ],
  layout: {
    columns: 4,  // Grid columns
    rows: [
      // Row 1: Two small cards
      { widgets: [0, 1], span: [1, 1] },
      // Row 2: Activity feed (2 cols) + Quick links (2 cols)
      { widgets: [2, 3], span: [2, 2] },
    ],
  },
}))
```

## Best Practices

### 1. Handle Loading States

```typescript
function MyWidget({ config }) {
  const [loading, setLoading] = useState(true);
  const [data, setData] = useState(null);

  if (loading) {
    return (
      <Card>
        <CardContent className="p-6">
          <Skeleton className="h-8 w-full" />
          <Skeleton className="mt-2 h-4 w-3/4" />
        </CardContent>
      </Card>
    );
  }

  return <Card>...</Card>;
}
```

### 2. Handle Errors

```typescript
function MyWidget({ config }) {
  const [error, setError] = useState<Error | null>(null);

  if (error) {
    return (
      <Card>
        <CardContent className="p-6">
          <p className="text-destructive">Failed to load data</p>
          <Button variant="outline" onClick={reload}>
            Retry
          </Button>
        </CardContent>
      </Card>
    );
  }

  return <Card>...</Card>;
}
```

### 3. Use Consistent Card Layout

```typescript
function MyWidget({ config }) {
  return (
    <Card>
      <CardHeader>
        <CardTitle>{config.title}</CardTitle>
        {config.description && (
          <CardDescription>{config.description}</CardDescription>
        )}
      </CardHeader>
      <CardContent>
        {/* Widget content */}
      </CardContent>
      {config.showFooter && (
        <CardFooter>
          {/* Actions */}
        </CardFooter>
      )}
    </Card>
  );
}
```

### 4. Make Data Fetching Configurable

```typescript
interface MyWidgetConfig {
  // Let users provide their own data fetcher
  fetchData: () => Promise<Data>;
  // Or provide options for built-in fetching
  collection?: string;
  query?: QueryOptions;
}
```
