# UI Components

QUESTPIE Admin exports UI components built with shadcn/ui for creating custom admin pages.

## Navigation Components

### AdminLink

Type-safe link component:

```tsx
import { AdminLink } from "@questpie/admin/client";

// Link to collection list
<AdminLink collection="posts">Posts</AdminLink>

// Link to collection item
<AdminLink collection="posts" id="post-123">Edit Post</AdminLink>

// Link to create form
<AdminLink collection="posts" action="create">New Post</AdminLink>

// Link to global
<AdminLink global="siteSettings">Settings</AdminLink>

// Link to dashboard
<AdminLink dashboard>Dashboard</AdminLink>
```

### CollectionLink

Shorthand for collection links:

```tsx
import { CollectionLink } from "@questpie/admin/client";

<CollectionLink collection="posts" />
<CollectionLink collection="posts" id="post-123" />
```

### GlobalLink

Shorthand for global links:

```tsx
import { GlobalLink } from "@questpie/admin/client";

<GlobalLink global="siteSettings" />
```

## Data Display

### Table

```tsx
import {
  Table,
  TableHeader,
  TableBody,
  TableRow,
  TableHead,
  TableCell,
} from "@questpie/admin/client";

<Table>
  <TableHeader>
    <TableRow>
      <TableHead>Title</TableHead>
      <TableHead>Status</TableHead>
    </TableRow>
  </TableHeader>
  <TableBody>
    {posts.map((post) => (
      <TableRow key={post.id}>
        <TableCell>{post.title}</TableCell>
        <TableCell>
          <Badge>{post.status}</Badge>
        </TableCell>
      </TableRow>
    ))}
  </TableBody>
</Table>
```

## Feedback

### Toast

Use `sonner` for toast notifications:

```tsx
import { toast } from "sonner";

// Success
toast.success("Changes saved");

// Error
toast.error("Failed to save");

// Loading
toast.loading("Saving...");

// Custom
toast("Custom message", {
  description: "More details here",
  action: {
    label: "Undo",
    onClick: () => handleUndo(),
  },
});
```

## Utility Hooks

### useMediaQuery

```tsx
import { useMediaQuery, useIsMobile, useIsDesktop } from "@questpie/admin/client";

function ResponsiveComponent() {
  const isMobile = useIsMobile();
  const isDesktop = useIsDesktop();
  const isLarge = useMediaQuery("(min-width: 1024px)");

  return isMobile ? <MobileView /> : <DesktopView />;
}
```

## Using Components in Custom Pages

```tsx
// routes/admin/dashboard.tsx
import {
  Card,
  CardHeader,
  CardTitle,
  CardContent,
  Button,
  Badge,
  useCollectionList,
  AdminLink,
} from "@questpie/admin/client";

export function CustomDashboard() {
  const { data } = useCollectionList("posts", { limit: 5 });

  return (
    <div className="grid gap-4 md:grid-cols-2">
      <Card>
        <CardHeader>
          <CardTitle>Recent Posts</CardTitle>
        </CardHeader>
        <CardContent>
          <ul className="space-y-2">
            {data?.docs.map((post) => (
              <li key={post.id} className="flex justify-between">
                <AdminLink collection="posts" id={post.id}>
                  {post.title}
                </AdminLink>
                <Badge>{post.status}</Badge>
              </li>
            ))}
          </ul>
          <Button asChild className="mt-4">
            <AdminLink collection="posts">View All</AdminLink>
          </Button>
        </CardContent>
      </Card>
    </div>
  );
}
```
